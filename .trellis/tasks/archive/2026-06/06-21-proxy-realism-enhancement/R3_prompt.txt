你是 Go 后端工程师，为 sub2api 实施 **R3：TLS 家族化配对指纹库（框架搭建）**。

## 上下文
- 仓库根 `/mnt/x/project/sub2api`，后端 `backend/`，分支 `feat/proxy-realism`。
- 目标：建立 `clientidentity` package，实现 family → identity snapshot 的解析，为后续 UA 联动（R1）与真实指纹数据采集留好接口。

## 关键设计点（来自 design.md，codex 审查已修正）

- **family 解析在上层 resolver**，不进 `tlsfingerprint/dialer.go`（保持低层执行）。
- resolver 返回**完整 snapshot**：`{headers, version fields, tls profile name}`。
- header / billing / identity 注入点**全部从同一 resolver 取值**，杜绝错配。
- 默认 family 行为 = 现有逻辑（claude 账号 = 现有 claude 指纹，codex 账号 = 现有 codex 指纹）。

// family.go
package clientidentity

type ClientFamily string

const (
	FamilyClaudeCLI     ClientFamily = "claude-cli"
	FamilyClaudeDesktop ClientFamily = "claude-desktop"
	FamilyCodexCLI      ClientFamily = "codex-cli"
	FamilyCodexDesktop  ClientFamily = "codex-desktop"
)

// snapshot.go
type IdentitySnapshot struct {
	Headers       map[string]string  // UA, X-Stainless-*, X-App, anthropic-beta 等
	VersionFields VersionFields      // cli/sdk 版本、cc_version、fp
	TLSProfileName string            // 对应 tlsfingerprint.Profile 的名称
}

type VersionFields struct {
	CLIVersion string
	SDKVersion string
	CCVersion  string  // 用于 billing cc_version=X.Y.Z.{fp}
	CCFP       string  // fingerprint 后缀
}

// resolver.go
type Resolver struct {
	// 内置家族预设（R1 UA 自动拉取后会用 atomic.Pointer 持有动态 snapshot）
	// 现在先用硬编码兜底
}

func NewResolver() *Resolver {
	return &Resolver{}
}

// Resolve 根据 account.extra.client_family 返回完整 snapshot
// 无 family 或 family 不识别 -> 返回 nil（调用方回退现有默认）
func (r *Resolver) Resolve(account *service.Account) *IdentitySnapshot {
	familyStr, ok := account.Extra["client_family"].(string)
	if !ok || familyStr == "" {
		return nil // 回退现有默认
	}

	family := ClientFamily(familyStr)
	switch family {
	case FamilyClaudeCLI:
		return r.claudeCLISnapshot()
	case FamilyClaudeDesktop:
		return r.claudeDesktopSnapshot()
	case FamilyCodexCLI:
		return r.codexCLISnapshot()
	case FamilyCodexDesktop:
		return r.codexDesktopSnapshot()
	default:
		return nil
	}
}

func (r *Resolver) claudeCLISnapshot() *IdentitySnapshot {
	// TODO(R1): 从 IdentityRegistry atomic.Pointer 读取动态版本
	// 现在先返回现有硬编码常量
	return &IdentitySnapshot{
		Headers: map[string]string{
			"User-Agent": "claude-cli/0.x.x (...)", // 占位，实际从 pkg/claude.DefaultHeaders 取
			// X-Stainless-*, X-App, anthropic-beta 等
		},
		VersionFields: VersionFields{
			CLIVersion: "0.x.x",
			SDKVersion: "0.y.y",
		},
		TLSProfileName: "claude-cli-default", // 对应 tlsfingerprint 已有 profile
	}
}

在需要 TLS profile 的地方（如 `account_usage_service.go:1165` 调用 `ResolveTLSProfile`），**先尝试 clientidentity resolver**：

```go
// account_usage_service.go（伪代码）
import "github.com/Wei-Shaw/sub2api/internal/pkg/clientidentity"

func (s *AccountUsageService) someMethod(..., account *service.Account) {
	// 1. 尝试 clientidentity resolver
	resolver := clientidentity.NewResolver()
	snapshot := resolver.Resolve(account)

	var tlsProfile string
	if snapshot != nil {
		tlsProfile = snapshot.TLSProfileName
	} else {
		// 2. 回退现有逻辑（无 family = 现有默认）
		tlsProfile = s.tlsFingerprintProfileService.ResolveTLSProfile(account)
	}

	// 3. 使用 tlsProfile 创建 dialer
	dialer := tlsfingerprint.NewDialer(tlsProfile, ...)
}
```

**关键**：这步只搭框架，**不改 header/billing 注入点**（那是 R1 的事）。R3 只负责 TLS profile 选择路径对接 clientidentity。

### B4. 单测
`backend/internal/pkg/clientidentity/resolver_test.go`：

```go
func TestResolve(t *testing.T) {
	resolver := NewResolver()

	// 1. account.extra.client_family = "claude-cli" -> 返回 claude-cli snapshot
	account := &service.Account{Extra: map[string]interface{}{"client_family": "claude-cli"}}
	snapshot := resolver.Resolve(account)
	assert.NotNil(t, snapshot)
	assert.Equal(t, "claude-cli-default", snapshot.TLSProfileName)

	// 2. 无 client_family -> 返回 nil（回退现有默认）
	account2 := &service.Account{Extra: map[string]interface{}{}}
	snapshot2 := resolver.Resolve(account2)
	assert.Nil(t, snapshot2)

	// 3. 不识别的 family -> 返回 nil
	account3 := &service.Account{Extra: map[string]interface{"client_family": "unknown"}}
	snapshot3 := resolver.Resolve(account3)
	assert.Nil(t, snapshot3)
}
```
