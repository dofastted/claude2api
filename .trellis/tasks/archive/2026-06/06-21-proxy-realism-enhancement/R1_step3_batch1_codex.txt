// resolveOpenAIUpstreamOriginator 改为从 registry 读取默认 originator
func (s *OpenAIGatewayService) resolveOpenAIUpstreamOriginator(c *gin.Context, isCodexCLI bool) string {
	// 1. 先读入站 originator（现有逻辑保持）
	originator := c.GetHeader("originator")
	if originator != "" {
		return originator
	}

	// 2. 官方客户端，从 registry 读取
	if isCodexCLI {
		codexSnapshot := s.identityRegistry.Get().Codex
		// codexSnapshot.Headers["originator"] 应为 "codex_cli_rs"
		if orig, ok := codexSnapshot.Headers["originator"]; ok && orig != "" {
			return orig
		}
		// 兜底（不应到这里，registry 初始化已设置）
		return "codex_cli_rs"
	}

	return "opencode"
}

// buildUpstreamRequestOpenAIPassthrough 改为从 registry 读取
func (s *OpenAIGatewayService) buildUpstreamRequestOpenAIPassthrough(...) {
	// 行3449: version = codexCLIVersion

	codexSnapshot := s.identityRegistry.Get().Codex
	version := codexSnapshot.VersionFields.CLIVersion

	// 行3461/3484/3488: user-agent = codexCLIUserAgent
	// 改为:
	userAgent := codexSnapshot.Headers["User-Agent"]

	// 行3461: originator = codex_cli_rs
	// 改为:
	originator := codexSnapshot.Headers["originator"]
	if originator == "" {
		originator = "codex_cli_rs" // 兜底
	}

	// ... 其余逻辑保持
}
