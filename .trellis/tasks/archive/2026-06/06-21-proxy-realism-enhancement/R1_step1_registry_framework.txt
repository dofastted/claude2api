## 关键设计点（来自 design.md，codex 审查已修正）

- **atomic 一致性**：一次更新 = 一个完整一致 snapshot（claude + codex 两族基线）；任一字段缺失则整体丢弃保留旧值。
- **联动落点全集**（暂不改造，只预留接口）：
  - Claude：`pkg/claude.DefaultHeaders` + `service/gateway_billing_block.go:73`(cc_version) + `service/identity_service.go:28` + `service/gateway_service.go:4322`
  - Codex：`service/openai_gateway_service.go:45/1268/71`(UA/originator/x-codex)
- **不在请求热路径联网**：fetch 在后台 goroutine，带超时/重试/频率限制。
- **开关与回滚**：config `gateway.ua_auto_fetch.{enabled(默认false),interval,sources}`；关闭=回退硬编码常量（今日行为）。

```go
package clientidentity

import "sync/atomic"

// Registry 持有当前生效的 claude + codex 两族完整 snapshot
type Registry struct {
	snapshotPtr atomic.Pointer[Snapshots]
}

// Snapshots 是两族的完整基线
type Snapshots struct {
	Claude ClaudeSnapshot
	Codex  CodexSnapshot
}

type ClaudeSnapshot struct {
	Headers       map[string]string  // UA, X-Stainless-*, X-App, anthropic-beta
	VersionFields VersionFields      // cli/sdk/cc_version/fp
	TLSProfileName string
}

type CodexSnapshot struct {
	Headers       map[string]string  // UA, originator, x-codex-*
	VersionFields VersionFields
	TLSProfileName string
}

func NewRegistry() *Registry {
	r := &Registry{}
	// 初始化为内置兜底（现有硬编码常量）
	r.snapshotPtr.Store(&Snapshots{
		Claude: r.defaultClaudeSnapshot(),
		Codex:  r.defaultCodexSnapshot(),
	})
	return r
}

// Get 只读 getter（原子）
func (r *Registry) Get() *Snapshots {
	return r.snapshotPtr.Load()
}

// Swap 原子更新（仅 VersionFetcher 调用）
func (r *Registry) Swap(snapshots *Snapshots) {
	r.snapshotPtr.Store(snapshots)
}

func (r *Registry) defaultClaudeSnapshot() ClaudeSnapshot {
	// TODO: 从 pkg/claude.DefaultHeaders / CLICurrentVersion 读取

		Headers: map[string]string{
			"User-Agent": "claude-code/...", // 占位

			CLIVersion: "0.x.x",
			SDKVersion: "0.y.y",
		},
		TLSProfileName: "claude-cli-default",
	}
}
```

### C2. VersionFetcher service（框架，不真联网）
在 `backend/internal/service/version_fetcher_service.go`：

```go
package service

import (
	"context"
	"time"
	"github.com/Wei-Shaw/sub2api/internal/pkg/clientidentity"
)

type VersionFetcherService struct {
	registry *clientidentity.Registry
	cfg      *config.Config
	stopCh   chan struct{}
}

func NewVersionFetcherService(registry *clientidentity.Registry, cfg *config.Config) *VersionFetcherService {
	return &VersionFetcherService{
		registry: registry,
		cfg:      cfg,
		stopCh:   make(chan struct{}),
	}
}

// Start 后台周期任务（仅当 cfg.Gateway.UAAutoFetch.Enabled 时）
func (s *VersionFetcherService) Start() {
	if !s.cfg.Gateway.UAAutoFetch.Enabled {
		return // 关闭开关 = 不启动，registry 保持内置兜底
	}

	interval := s.cfg.Gateway.UAAutoFetch.Interval
	if interval == 0 {
		interval = 1 * time.Hour // 默认间隔
	}

	go func() {
		ticker := time.NewTicker(interval)
		defer ticker.Stop()

		for {
			select {
			case <-ticker.C:
				s.fetchAndUpdate()
			case <-s.stopCh:
				return
			}
		}
	}()
}

func (s *VersionFetcherService) Stop() {
	close(s.stopCh)
}

func (s *VersionFetcherService) fetchAndUpdate() {
	ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
	defer cancel()

	// TODO: 真实拉取逻辑（npm @anthropic-ai/claude-code、github openai/codex）
	// 现在只是框架，不真联网
	claudeVersion, claudeSDKVersion, codexVersion := s.fetchVersions(ctx)

	if claudeVersion == "" || codexVersion == "" {
		// 失败丢弃保留旧值
		return
	}

	// 组装完整一致 snapshot（任一字段缺失则整体丢弃）
	snapshots := &clientidentity.Snapshots{
		Claude: clientidentity.ClaudeSnapshot{
			Headers: s.buildClaudeHeaders(claudeVersion, claudeSDKVersion),
			VersionFields: clientidentity.VersionFields{
				CLIVersion: claudeVersion,
				SDKVersion: claudeSDKVersion,
			},
			TLSProfileName: "claude-cli-default",
		},
		Codex: clientidentity.CodexSnapshot{
			Headers: s.buildCodexHeaders(codexVersion),
			VersionFields: clientidentity.VersionFields{
				CLIVersion: codexVersion,
			},
			TLSProfileName: "codex-cli-default",
		},
	}

	// 原子 swap
	s.registry.Swap(snapshots)
}

func (s *VersionFetcherService) fetchVersions(ctx context.Context) (claudeVersion, claudeSDKVersion, codexVersion string) {
	// TODO: 实际 npm/github API 调用
	// 占位：直接返回空（触发丢弃）
	return "", "", ""
}

func (s *VersionFetcherService) buildClaudeHeaders(cliVer, sdkVer string) map[string]string {
	// TODO: 按 design 完整构造 UA / X-Stainless-* / X-App / anthropic-beta

		"User-Agent": "claude-code/" + cliVer,
	}
}

func (s *VersionFetcherService) buildCodexHeaders(cliVer string) map[string]string {
```

```go
func TestRegistryAtomicSwap(t *testing.T) {
	registry := NewRegistry()
	initial := registry.Get()

	newSnapshots := &Snapshots{
		Claude: ClaudeSnapshot{VersionFields: VersionFields{CLIVersion: "1.0.0"}},
	}
	registry.Swap(newSnapshots)

	assert.Equal(t, "1.0.0", registry.Get().Claude.VersionFields.CLIVersion)
	assert.NotEqual(t, initial, registry.Get()) // 指针已变
}
```
