| 字段 | 目标值 | 必要性 |
|------|--------|--------|
| `openai_oauth_responses_websockets_v2_enabled` | `true` | **必须**（分类型 enabled 优先级最高） |
| `openai_oauth_responses_websockets_v2_mode` | `ctx_pool` | OAuth 模式 |
| `openai_apikey_responses_websockets_v2_enabled` | `true` | **必须**（分类型 enabled 优先级最高） |
| `openai_apikey_responses_websockets_v2_mode` | `ctx_pool` | API Key 模式 |
| `responses_websockets_v2_enabled` | `true` | 兼容总开关 |
| `openai_ws_enabled` | `true` | 历史总开关 |
| `openai_ws_allow_store_recovery` | `true` | store 恢复 |
| `openai_ws_force_http` | `false` | 不强制降级 |

```go
// EnableAllOpenAIWS 把 OpenAI 账号的 ws 相关字段全部写为目标态
func (s *AccountService) EnableAllOpenAIWS(ctx context.Context, accountID int64) error {
	// 1. 读取账号
	account, err := s.GetByID(ctx, accountID)
	if err != nil { return err }

	// 2. 校验是 OpenAI 账号
	if account.Platform != "openai" {
		return fmt.Errorf("account %d is not OpenAI platform", accountID)
	}

	// 3. 前置校验：检查全局 gateway.openai_ws.responses_websockets_v2 是否被禁用
	// 若被禁用，账号 extra 无法单独开启 -> 返回明确提示
	globalWSEnabled := s.cfg.Gateway.OpenAIWS.ResponsesWebSocketsV2 // 假设路径，实际需确认
	if !globalWSEnabled {
		return fmt.Errorf("global gateway.openai_ws.responses_websockets_v2 is disabled, cannot enable for account")
	}

	// 4. 准备 8 字段
	updates := map[string]interface{}{
		"openai_oauth_responses_websockets_v2_enabled":  true,
		"openai_oauth_responses_websockets_v2_mode":      "ctx_pool",
		"openai_apikey_responses_websockets_v2_enabled": true,
		"openai_apikey_responses_websockets_v2_mode":     "ctx_pool",
		"responses_websockets_v2_enabled":                true,
		"openai_ws_enabled":                              true,
		"openai_ws_allow_store_recovery":                 true,
		"openai_ws_force_http":                           false,
	}

	// 5. 合并到 extra
	extra := account.Extra
	if extra == nil {
		extra = make(map[string]interface{})
	}
	for k, v := range updates {
		extra[k] = v
	}

	// 6. 更新数据库
	return s.repo.UpdateExtra(ctx, accountID, extra)
}

// ResetOpenAIWS 删除这 8 个字段，回归默认
func (s *AccountService) ResetOpenAIWS(ctx context.Context, accountID int64) error {
	account, err := s.GetByID(ctx, accountID)
	if err != nil { return err }
	if account.Platform != "openai" {
		return fmt.Errorf("account %d is not OpenAI platform", accountID)
	}

	extra := account.Extra
	if extra == nil { return nil }

	keys := []string{
		"openai_oauth_responses_websockets_v2_enabled",
		"openai_oauth_responses_websockets_v2_mode",
		"openai_apikey_responses_websockets_v2_enabled",
		"openai_apikey_responses_websockets_v2_mode",
		"responses_websockets_v2_enabled",
		"openai_ws_enabled",
		"openai_ws_allow_store_recovery",
		"openai_ws_force_http",
	}
	for _, k := range keys {
		delete(extra, k)
	}

	return s.repo.UpdateExtra(ctx, accountID, extra)
}
```

### A2. admin handler + 路由
在 `backend/internal/handler/admin/account_handler.go` 或新建 `openai_ws_handler.go`：

```go
// EnableAllOpenAIWS godoc
// @Summary 一键开启账号的 OpenAI WebSocket 全部功能
// @Tags Admin - Accounts
// @Accept json
// @Produce json
// @Param id path int true "Account ID"
// @Success 200 {object} response.Success
// @Failure 400 {object} response.Error
// @Router /admin/accounts/{id}/openai-ws/enable-all [post]
func (h *AccountHandler) EnableAllOpenAIWS(c *gin.Context) {
	accountID, err := parseIDParam(c)
	if err != nil { AbortWithError(c, 400, "INVALID_ID", err.Error()); return }

	if err := h.accountService.EnableAllOpenAIWS(c.Request.Context(), accountID); err != nil {
		AbortWithError(c, 400, "ENABLE_FAILED", err.Error())
		return
	}

	c.JSON(200, response.Success{Message: "OpenAI WS features enabled"})
}

// ResetOpenAIWS godoc
// @Summary 重置账号的 OpenAI WebSocket 配置（删除 8 字段）
// @Tags Admin - Accounts
// @Accept json
// @Produce json
// @Param id path int true "Account ID"
// @Success 200 {object} response.Success
// @Failure 400 {object} response.Error
// @Router /admin/accounts/{id}/openai-ws/reset [post]
func (h *AccountHandler) ResetOpenAIWS(c *gin.Context) {
	accountID, err := parseIDParam(c)
	if err != nil { AbortWithError(c, 400, "INVALID_ID", err.Error()); return }

	if err := h.accountService.ResetOpenAIWS(c.Request.Context(), accountID); err != nil {
		AbortWithError(c, 400, "RESET_FAILED", err.Error())
		return
	}

	c.JSON(200, response.Success{Message: "OpenAI WS configuration reset"})
}
```
