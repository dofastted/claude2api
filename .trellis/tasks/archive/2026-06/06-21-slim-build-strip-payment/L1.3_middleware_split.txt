### 4. slim 实现重点
slim 版中间件伪代码：
```go
func APIKeyAuthWithSubscription(...) gin.HandlerFunc {
	return func(c *gin.Context) {
		// 1. 解析 API key
		key, err := extractKey(c)
		if err != nil { c.AbortWithStatusJSON(401, ...); return }

		// 2. 验证 key 真伪
		apiKey, err := apiKeyService.GetByKey(ctx, key)
		if err != nil || apiKey == nil { c.AbortWithStatusJSON(401, ...); return }

		// 3. 用户状态
		user, err := userService.GetByID(ctx, apiKey.UserID)
		if err != nil || user.Status != "active" { c.AbortWithStatusJSON(403, ...); return }

		// 4. group 可用性
		group, err := groupService.GetByID(ctx, apiKey.GroupID)
		if err != nil || !group.IsActive { c.AbortWithStatusJSON(403, ...); return }

		// 5. IP 限制
		if !checkIPAllowed(c.ClientIP(), group.AllowedIPs) { c.AbortWithStatusJSON(403, ...); return }

		// 6. slim: 跳过订阅/余额/配额，直接放行
		c.Set("api_key", apiKey)
		c.Set("user", user)
		c.Set("group", group)
		c.Next()
	}
}
```

**关键**：不要在参数里传 nil `subscriptionService` 并依赖原逻辑回退，因为回退逻辑仍会做余额检查。slim 要**显式重写**，跳过整个订阅/余额段。
